    2. Create job_progress_updates table:

       - created_at timestamp default CURRENT_TIMESTAMP

       - job_id varchar(255) foreign key to jobs(id) ON DELETE CASCADE
       - level varchar(20) not null ('info', 'warn', 'error', 'debug')
       - message text not null
       - source varchar(100) (optional component name)
       - created_at timestamp default CURRENT_TIMESTAMP
       - Index on (job_id, created_at DESC)

    4. Create index on jobs.last_heartbeat for stale job queries:
