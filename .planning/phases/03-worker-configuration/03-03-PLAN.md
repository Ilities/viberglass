          - For 'docker':
            for (const file of payload.instructionFiles) {
              try {

                this.instructionFiles.set(file.fileType, content)
              } catch (error) {
                this.logger.warn('Failed to read mounted config file', { path: file.mountPath, error })
              }
            }
