```typescript
import crypto from 'crypto';
import axios from 'axios';
import {
  WebhookProvider,
  WebhookEvent,
  ProviderConfig,
  WebhookSetupInstructions,
  GitHubIssuePayload,
} from '../types';
import { createChildLogger } from '../../config/logger';

const logger = createChildLogger({ component: 'GitHubWebhookProvider' });

export class GitHubWebhookProvider implements WebhookProvider {
  readonly name = 'GitHubWebhookProvider';
  readonly providerType = 'github' as const;

  /**
   * Validate GitHub webhook signature using HMAC-SHA256
   * CRITICAL: Uses timing-safe comparison to prevent timing attacks
   */
  validateSignature(payload: string, signature: string, secret: string): boolean {
    if (!signature || !secret) {
      logger.warn('Missing signature or secret');
      return false;
    }

    const expectedSignature = `sha256=${crypto
      .createHmac('sha256', secret)
      .update(payload, 'utf8')
      .digest('hex')}`;

    try {
      // Timing-safe comparison prevents timing attacks
      return crypto.timingSafeEqual(
        Buffer.from(signature),
        Buffer.from(expectedSignature)
      );
    } catch (error) {
      // Buffer length mismatch throws - signature invalid
      logger.warn('Signature comparison failed', { error: (error as Error).message });
      return false;
    }
  }

  /**
   * Parse GitHub webhook payload into normalized WebhookEvent
   * Returns null for events we don't process (non-issues, no @bot mention)
   */
  parseEvent(headers: Record<string, string>, payload: unknown): WebhookEvent | null {
    const eventType = headers['x-github-event'];
    const deliveryId = headers['x-github-delivery'];

    if (!eventType || !deliveryId) {
      logger.warn('Missing required headers', { eventType, deliveryId });
      return null;
    }

    // Only process issues and issue_comment events
    if (eventType !== 'issues' && eventType !== 'issue_comment') {
      logger.debug('Ignoring non-issue event', { eventType });
      return null;
    }

    const data = payload as GitHubIssuePayload;

    if (!data.issue || !data.repository || !data.sender) {
      logger.warn('Invalid payload structure');
      return null;
    }

    return {
      deliveryId,
      eventType,
      action: data.action,
      repository: {
        owner: data.repository.owner.login,
        name: data.repository.name,
        fullName: data.repository.full_name,
      },
      issue: {
        number: data.issue.number,
        title: data.issue.title,
        body: data.issue.body || '',
        labels: data.issue.labels?.map(l => l.name) || [],
        url: data.issue.html_url,
      },
      comment: data.comment ? {
        id: data.comment.id,
        body: data.comment.body,
        author: data.comment.user.login,
      } : undefined,
      sender: {
        login: data.sender.login,
      },
      rawPayload: payload,
    };
  }

  /**
   * Post a comment on a GitHub issue
   */
  async postComment(config: ProviderConfig, issueNumber: number, body: string): Promise<void> {
    const url = `https://api.github.com/repos/${config.owner}/${config.repo}/issues/${issueNumber}/comments`;

    try {
      await axios.post(url, { body }, {
        headers: {
          Authorization: `token ${config.token}`,
          Accept: 'application/vnd.github.v3+json',
          'User-Agent': 'viberator/1.0',
        },
      });

      logger.info('Posted comment', { owner: config.owner, repo: config.repo, issueNumber });
    } catch (error) {
      logger.error('Failed to post comment', {
        error: (error as Error).message,
        issueNumber,
      });
      throw error;
    }
  }

  /**
   * Update labels on a GitHub issue
   */
  async updateLabels(
    config: ProviderConfig,
    issueNumber: number,
    add: string[],
    remove: string[]
  ): Promise<void> {
    const baseUrl = `https://api.github.com/repos/${config.owner}/${config.repo}/issues/${issueNumber}/labels`;
    const headers = {
      Authorization: `token ${config.token}`,
      Accept: 'application/vnd.github.v3+json',
      'User-Agent': 'viberator/1.0',
    };

    try {
      // Add labels
      if (add.length > 0) {
        await axios.post(baseUrl, { labels: add }, { headers });
        logger.debug('Added labels', { issueNumber, labels: add });
      }

      // Remove labels one by one (GitHub API limitation)
      for (const label of remove) {
        try {
          await axios.delete(`${baseUrl}/${encodeURIComponent(label)}`, { headers });
          logger.debug('Removed label', { issueNumber, label });
        } catch (error) {
          // Label might not exist, log and continue
          logger.warn('Failed to remove label', { issueNumber, label, error: (error as Error).message });
        }
      }
    } catch (error) {
      logger.error('Failed to update labels', {
        error: (error as Error).message,
        issueNumber,
      });
      throw error;
    }
  }

  /**
   * Get setup instructions for GitHub webhook configuration
   */
  getSetupInstructions(webhookUrl: string): WebhookSetupInstructions {
    return {
      title: 'GitHub Webhook Setup',
      steps: [
        '1. Go to your repository Settings > Webhooks > Add webhook',
        '2. Set the Payload URL to the URL shown below',
        '3. Set Content type to "application/json"',
        '4. Enter the webhook secret (shown below or paste your own)',
        '5. Select "Let me select individual events" and check: Issues, Issue comments',
        '6. Ensure "Active" is checked and save',
      ],
      payloadUrl: webhookUrl,
      contentType: 'application/json',
      events: ['issues', 'issue_comment'],
      secretInstructions: 'Copy the secret below or generate your own. Keep it secure!',
    };
  }
}

/**
 * Detect if body contains @bot mention
 */
export function detectBotMention(body: string, botUsername: string): boolean {
  if (!body || !botUsername) return false;
  const pattern = new RegExp(`@${botUsername}\\b`, 'i');
  return pattern.test(body);
}

/**
 * Extract intent/command after @bot mention
 */
export function extractBotIntent(body: string, botUsername: string): string | null {
  if (!body || !botUsername) return null;
  const pattern = new RegExp(`@${botUsername}\\s+(.+?)(?:\\n|$)`, 'i');
  const match = body.match(pattern);
  return match ? match[1].trim() : null;
}
```

```typescript
import { WebhookProvider } from './types';
import { GitHubWebhookProvider } from './providers/GitHubWebhookProvider';
import { createChildLogger } from '../config/logger';

const logger = createChildLogger({ component: 'WebhookProviderFactory' });

export type ProviderType = 'github' | 'gitlab' | 'bitbucket';

/**
 * Factory for creating webhook providers
 * Follows the same pattern as CredentialProviderFactory
 */
export class WebhookProviderFactory {
  private providers: Map<ProviderType, WebhookProvider> = new Map();

  constructor() {
    // Register built-in providers
    this.registerProvider(new GitHubWebhookProvider());
    // Future: GitLabWebhookProvider, BitbucketWebhookProvider
  }

  /**
   * Register a provider instance
   */
  registerProvider(provider: WebhookProvider): void {
    this.providers.set(provider.providerType, provider);
    logger.info('Registered webhook provider', { type: provider.providerType, name: provider.name });
  }

  /**
   * Get provider by type
   * Throws if provider not found
   */
  getProvider(type: ProviderType): WebhookProvider {
    const provider = this.providers.get(type);
    if (!provider) {
      throw new Error(`Unknown webhook provider type: ${type}`);
    }
    return provider;
  }

  /**
   * Check if provider type is supported
   */
  hasProvider(type: ProviderType): boolean {
    return this.providers.has(type);
  }

  /**
   * Get all registered provider types
   */
  getRegisteredTypes(): ProviderType[] {
    return Array.from(this.providers.keys());
  }
}

// Singleton instance
let factoryInstance: WebhookProviderFactory | null = null;

/**
 * Get singleton factory instance
 */
export function getWebhookProviderFactory(): WebhookProviderFactory {
  if (!factoryInstance) {
    factoryInstance = new WebhookProviderFactory();
  }
  return factoryInstance;
}

/**
 * Reset factory (for testing)
 */
export function resetWebhookProviderFactory(): void {
  factoryInstance = null;
}
```
