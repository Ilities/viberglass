```typescript
import crypto from 'crypto';
import { WebhookEvent, WebhookConfig, ProviderType } from './types';
import { getWebhookProviderFactory } from './WebhookProviderFactory';
import { WebhookDeduplicationService } from './WebhookDeduplicationService';
import { WebhookConfigDAO } from './WebhookConfigDAO';
import { WebhookFailedEventDAO } from './WebhookFailedEventDAO';
import { detectBotMention } from './providers/GitHubWebhookProvider';
import { TicketDAO } from '../persistence/ticketing/TicketDAO';
import { JobService } from '../services/JobService';
import { WorkerExecutionService } from '../workers/WorkerExecutionService';
import { ClankerDAO } from '../persistence/clanker/ClankerDAO';
import { CredentialProviderFactory } from '../credentials/CredentialProviderFactory';
import { createChildLogger } from '../config/logger';
import db from '../persistence/config/database';

const logger = createChildLogger({ component: 'WebhookService' });

export interface WebhookProcessResult {
  success: boolean;
  action: 'created_ticket' | 'created_job' | 'skipped_duplicate' | 'skipped_no_mention' | 'skipped_no_config' | 'failed';
  ticketId?: string;
  jobId?: string;
  error?: string;
}

/**
 * Orchestrates webhook processing:
 * 1. Validate signature
 * 2. Parse event
 * 3. Check deduplication
 * 4. Find project config
 * 5. Check @bot mention (for comments)
 * 6. Create ticket
 * 7. Optionally trigger auto-execute
 */
export class WebhookService {
  private providerFactory = getWebhookProviderFactory();
  private deduplicationService = new WebhookDeduplicationService();
  private configDAO = new WebhookConfigDAO();
  private failedEventDAO = new WebhookFailedEventDAO();
  private ticketDAO = new TicketDAO();
  private jobService = new JobService();
  private workerExecutionService = new WorkerExecutionService();
  private clankerDAO = new ClankerDAO();
  private credentialProvider: CredentialProviderFactory;

  constructor(credentialProvider?: CredentialProviderFactory) {
    this.credentialProvider = credentialProvider || new CredentialProviderFactory();
  }

  /**
   * Process incoming webhook
   * Returns 200 quickly to avoid timeout, processes synchronously
   */
  async processWebhook(
    providerType: ProviderType,
    headers: Record<string, string>,
    rawBody: string,
    payload: unknown
  ): Promise<WebhookProcessResult> {
    const provider = this.providerFactory.getProvider(providerType);

    // Step 1: Parse event to get basic info (for logging/error storage)
    const event = provider.parseEvent(headers, payload);

    try {
      // Step 2: Find config by repository
      if (!event) {
        logger.debug('Event not actionable (non-issue or missing data)');
        return { success: true, action: 'skipped_no_config' };
      }

      const configResult = await this.configDAO.getConfigByRepository(
        event.repository.fullName,
        providerType
      );

      if (!configResult) {
        logger.info('No webhook config for repository', {
          repository: event.repository.fullName,
        });
        return { success: true, action: 'skipped_no_config' };
      }

      const { config, projectId } = configResult;

      // Step 3: Validate signature
      const secret = await this.credentialProvider.get('default', config.secretKey);
      if (!secret) {
        logger.error('Webhook secret not found', { secretKey: config.secretKey });
        return { success: false, action: 'failed', error: 'Webhook secret not configured' };
      }

      const signature = this.getSignatureHeader(providerType, headers);
      if (!provider.validateSignature(rawBody, signature, secret)) {
        logger.warn('Invalid webhook signature', {
          deliveryId: event.deliveryId,
          repository: event.repository.fullName,
        });
        return { success: false, action: 'failed', error: 'Invalid signature' };
      }

      // Step 4: Check deduplication
      if (await this.deduplicationService.isDuplicate(event)) {
        logger.info('Duplicate webhook skipped', { deliveryId: event.deliveryId });
        return { success: true, action: 'skipped_duplicate' };
      }

      // Step 5: Check @bot mention (for comments)
      if (event.eventType === 'issue_comment' && config.botUsername) {
        const body = event.comment?.body || '';
        if (!detectBotMention(body, config.botUsername)) {
          logger.debug('Skipping comment without @bot mention', {
            deliveryId: event.deliveryId,
          });
          return { success: true, action: 'skipped_no_mention' };
        }
      }

      // Step 6: Create ticket
      const ticket = await this.createTicketFromEvent(event, projectId, config);

      // Mark as processed (after successful ticket creation)
      await this.deduplicationService.markProcessed(event);

      // Step 7: Auto-execute if configured
      if (config.autoExecute && config.defaultClankerId) {
        const jobResult = await this.triggerAutoExecute(
          ticket.id,
          projectId,
          config.defaultClankerId,
          event
        );

        if (jobResult) {
          return {
            success: true,
            action: 'created_job',
            ticketId: ticket.id,
            jobId: jobResult.jobId,
          };
        }
      }

      return { success: true, action: 'created_ticket', ticketId: ticket.id };

    } catch (error) {
      const errorMessage = (error as Error).message;
      logger.error('Webhook processing failed', {
        error: errorMessage,
        deliveryId: event?.deliveryId,
      });

      // Store failed event for manual retry
      await this.failedEventDAO.storeFailedEvent({
        projectId: null, // May not have found project yet
        providerType,
        deliveryId: event?.deliveryId || null,
        eventType: event?.eventType || null,
        headers,
        payload,
        errorMessage,
      });

      return { success: false, action: 'failed', error: errorMessage };
    }
  }

  /**
   * Retry a failed webhook event
   */
  async retryFailedEvent(failedEventId: string): Promise<WebhookProcessResult> {
    const failedEvent = await this.failedEventDAO.getFailedEvent(failedEventId);
    if (!failedEvent) {
      return { success: false, action: 'failed', error: 'Failed event not found' };
    }

    // Mark as retried before processing
    await this.failedEventDAO.markRetried(failedEventId);

    // Reconstruct headers and process
    const result = await this.processWebhook(
      failedEvent.providerType as ProviderType,
      failedEvent.headers,
      JSON.stringify(failedEvent.payload), // Re-stringify for signature (may fail if original was different)
      failedEvent.payload
    );

    // If successful, delete the failed event record
    if (result.success && result.action !== 'failed') {
      await this.failedEventDAO.deleteFailedEvent(failedEventId);
    }

    return result;
  }

  /**
   * Create ticket from webhook event
   */
  private async createTicketFromEvent(
    event: WebhookEvent,
    projectId: string,
    config: WebhookConfig
  ): Promise<{ id: string }> {
    // Determine ticket content based on event type
    const title = event.eventType === 'issue_comment'
      ? `Re: ${event.issue?.title || 'GitHub Comment'}`
      : event.issue?.title || 'GitHub Issue';

    const description = event.eventType === 'issue_comment'
      ? event.comment?.body || ''
      : event.issue?.body || '';

    // Create ticket using existing TicketDAO
    // Note: TicketDAO.createTicket expects full CreateTicketRequest with media assets
    // For webhooks, we use a simpler insert
    const ticketId = crypto.randomUUID();
    const now = new Date();

    await db
      .insertInto('tickets')
      .values({
        id: ticketId,
        project_id: projectId,
        timestamp: now,
        title,
        description,
        severity: 'medium', // Default for webhooks
        category: 'webhook',
        metadata: JSON.stringify({
          source: 'webhook',
          provider: config.providerType,
          repository: event.repository.fullName,
          issueNumber: event.issue?.number,
          issueUrl: event.issue?.url,
          commentId: event.comment?.id,
          sender: event.sender.login,
          deliveryId: event.deliveryId,
        }),
        annotations: JSON.stringify([]),
        ticket_system: 'github' as const,
        auto_fix_requested: config.autoExecute,
        external_ticket_id: event.issue?.number?.toString() || null,
        external_ticket_url: event.issue?.url || null,
        created_at: now,
        updated_at: now,
      })
      .execute();

    logger.info('Created ticket from webhook', {
      ticketId,
      projectId,
      issueNumber: event.issue?.number,
    });

    return { id: ticketId };
  }

  /**
   * Trigger auto-execution for a ticket
   */
  private async triggerAutoExecute(
    ticketId: string,
    projectId: string,
    clankerId: string,
    event: WebhookEvent
  ): Promise<{ jobId: string } | null> {
    try {
      // Get clanker
      const clanker = await this.clankerDAO.getClankerById(clankerId);
      if (!clanker) {
        logger.warn('Clanker not found for auto-execute', { clankerId });
        return null;
      }

      // Get project for repository URL
      const project = await db
        .selectFrom('projects')
        .select(['id', 'repository_url'])
        .where('id', '=', projectId)
        .executeTakeFirst();

      if (!project?.repository_url) {
        logger.warn('Project has no repository URL', { projectId });
        return null;
      }

      // Create job data
      const jobData = {
        id: crypto.randomUUID(),
        tenantId: 'default', // TODO: Get from project/config
        repository: project.repository_url,
        task: event.issue?.body || event.comment?.body || 'Fix issue from webhook',
        branch: null,
        baseBranch: 'main',
        context: {
          issueNumber: event.issue?.number,
          issueTitle: event.issue?.title,
          issueUrl: event.issue?.url,
        },
        settings: {},
        timestamp: Date.now(),
      };

      // Submit job
      const { jobId } = await this.jobService.submitJob(jobData, {
        ticketId,
        clankerId,
      });

      // Execute job
      await this.workerExecutionService.executeJob(jobData, clanker as any);

      logger.info('Auto-execute triggered', { jobId, ticketId, clankerId });

      return { jobId };
    } catch (error) {
      logger.error('Auto-execute failed', {
        error: (error as Error).message,
        ticketId,
        clankerId,
      });
      return null;
    }
  }

  /**
   * Get signature header based on provider type
   */
  private getSignatureHeader(
    providerType: ProviderType,
    headers: Record<string, string>
  ): string {
    switch (providerType) {
      case 'github':
        return headers['x-hub-signature-256'] || '';
      case 'gitlab':
        return headers['x-gitlab-token'] || '';
      case 'bitbucket':
        return headers['x-hub-signature'] || '';
      default:
        return '';
    }
  }
}
```

```typescript
import express from 'express';
import { WebhookService } from '../../webhooks/WebhookService';
import { WebhookConfigDAO } from '../../webhooks/WebhookConfigDAO';
import { WebhookFailedEventDAO } from '../../webhooks/WebhookFailedEventDAO';
import { getWebhookProviderFactory, ProviderType } from '../../webhooks';
import logger from '../../config/logger';

const router = express.Router();
const webhookService = new WebhookService();
const configDAO = new WebhookConfigDAO();
const failedEventDAO = new WebhookFailedEventDAO();

/**
 * Middleware to capture raw body for signature verification
 * Express.json() re-serializes the body, which breaks GitHub signature validation
 * This captures the raw body before parsing
 */
router.use('/:provider', express.json({
  verify: (req: express.Request, _res, buf) => {
    (req as any).rawBody = buf.toString('utf8');
  }
}));

/**
 * Provider-agnostic webhook endpoint
 * POST /api/webhooks/:provider (github, gitlab, bitbucket)
 */
router.post('/:provider', async (req, res) => {
  const providerType = req.params.provider as ProviderType;

  // Validate provider type
  const factory = getWebhookProviderFactory();
  if (!factory.hasProvider(providerType)) {
    logger.warn('Unknown webhook provider', { provider: providerType });
    return res.status(400).json({ error: `Unknown provider: ${providerType}` });
  }

  // Get raw body (captured by verify middleware)
  const rawBody = (req as any).rawBody as string;
  if (!rawBody) {
    logger.error('Raw body not captured');
    return res.status(500).json({ error: 'Internal error: raw body not captured' });
  }

  // Normalize headers to lowercase (Express normalizes, but be explicit)
  const headers: Record<string, string> = {};
  for (const [key, value] of Object.entries(req.headers)) {
    if (typeof value === 'string') {
      headers[key.toLowerCase()] = value;
    } else if (Array.isArray(value)) {
      headers[key.toLowerCase()] = value[0];
    }
  }

  // Process webhook
  const result = await webhookService.processWebhook(
    providerType,
    headers,
    rawBody,
    req.body
  );

  // Return 200 for all cases (even skipped) to prevent retries
  // Only return 4xx for actual client errors (bad signature handled internally)
  if (result.action === 'failed' && result.error?.includes('signature')) {
    return res.status(401).json({
      error: 'Unauthorized',
      message: 'Invalid webhook signature',
    });
  }

  res.status(200).json({
    success: result.success,
    action: result.action,
    ticketId: result.ticketId,
    jobId: result.jobId,
  });
});

/**
 * Get webhook configuration for a project
 * GET /api/webhooks/config/:projectId
 */
router.get('/config/:projectId', async (req, res) => {
  try {
    const configs = await configDAO.getConfigsByProject(req.params.projectId);
    res.json({ configs });
  } catch (error) {
    logger.error('Failed to get webhook configs', { error: (error as Error).message });
    res.status(500).json({ error: 'Failed to get webhook configs' });
  }
});

/**
 * Create/update webhook configuration
 * PUT /api/webhooks/config/:projectId/:provider
 */
router.put('/config/:projectId/:provider', async (req, res) => {
  try {
    const { projectId, provider } = req.params;
    const { secretKey, botUsername, autoExecute, defaultClankerId } = req.body;

    if (!secretKey) {
      return res.status(400).json({ error: 'secretKey is required' });
    }

    const config = await configDAO.upsertConfig({
      projectId,
      providerType: provider as ProviderType,
      secretKey,
      botUsername: botUsername || null,
      autoExecute: autoExecute ?? false,
      defaultClankerId: defaultClankerId || null,
    });

    res.json({ config });
  } catch (error) {
    logger.error('Failed to save webhook config', { error: (error as Error).message });
    res.status(500).json({ error: 'Failed to save webhook config' });
  }
});

/**
 * Delete webhook configuration
 * DELETE /api/webhooks/config/:projectId/:provider
 */
router.delete('/config/:projectId/:provider', async (req, res) => {
  try {
    const { projectId, provider } = req.params;
    const deleted = await configDAO.deleteConfig(projectId, provider as ProviderType);

    if (!deleted) {
      return res.status(404).json({ error: 'Config not found' });
    }

    res.json({ success: true });
  } catch (error) {
    logger.error('Failed to delete webhook config', { error: (error as Error).message });
    res.status(500).json({ error: 'Failed to delete webhook config' });
  }
});

/**
 * Get webhook setup instructions
 * GET /api/webhooks/setup/:provider
 */
router.get('/setup/:provider', (req, res) => {
  try {
    const providerType = req.params.provider as ProviderType;
    const factory = getWebhookProviderFactory();

    if (!factory.hasProvider(providerType)) {
      return res.status(400).json({ error: `Unknown provider: ${providerType}` });
    }

    const provider = factory.getProvider(providerType);
    const baseUrl = process.env.WEBHOOK_BASE_URL || `${req.protocol}://${req.get('host')}`;
    const webhookUrl = `${baseUrl}/api/webhooks/${providerType}`;

    const instructions = provider.getSetupInstructions(webhookUrl);
    res.json({ instructions });
  } catch (error) {
    logger.error('Failed to get setup instructions', { error: (error as Error).message });
    res.status(500).json({ error: 'Failed to get setup instructions' });
  }
});

/**
 * List failed webhook events
 * GET /api/webhooks/failed?projectId=xxx
 */
router.get('/failed', async (req, res) => {
  try {
    const projectId = req.query.projectId as string | undefined;
    const events = await failedEventDAO.getFailedEvents(projectId || null);
    res.json({ events });
  } catch (error) {
    logger.error('Failed to get failed events', { error: (error as Error).message });
    res.status(500).json({ error: 'Failed to get failed events' });
  }
});

/**
 * Retry a failed webhook event
 * POST /api/webhooks/failed/:id/retry
 */
router.post('/failed/:id/retry', async (req, res) => {
  try {
    const result = await webhookService.retryFailedEvent(req.params.id);
    res.json(result);
  } catch (error) {
    logger.error('Failed to retry webhook', { error: (error as Error).message });
    res.status(500).json({ error: 'Failed to retry webhook' });
  }
});

/**
 * Delete a failed webhook event
 * DELETE /api/webhooks/failed/:id
 */
router.delete('/failed/:id', async (req, res) => {
  try {
    const deleted = await failedEventDAO.deleteFailedEvent(req.params.id);
    if (!deleted) {
      return res.status(404).json({ error: 'Event not found' });
    }
    res.json({ success: true });
  } catch (error) {
    logger.error('Failed to delete failed event', { error: (error as Error).message });
    res.status(500).json({ error: 'Failed to delete failed event' });
  }
});

/**
 * Webhook status endpoint (legacy compatibility)
 * GET /api/webhooks/status
 */
router.get('/status', async (_req, res) => {
  try {
    const factory = getWebhookProviderFactory();
    res.json({
      status: 'healthy',
      providers: factory.getRegisteredTypes(),
    });
  } catch (error) {
    logger.error('Failed to get webhook status', { error: (error as Error).message });
    res.status(500).json({ error: 'Failed to get status' });
  }
});

export default router;
```
