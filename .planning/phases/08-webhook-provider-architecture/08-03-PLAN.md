```typescript
import crypto from 'crypto';
import db from '../persistence/config/database';
import { WebhookEvent } from './types';
import { createChildLogger } from '../config/logger';

const logger = createChildLogger({ component: 'WebhookDeduplicationService' });

/**
 * Handles webhook deduplication using content-based SHA-256 hashing
 * with a 24-hour TTL window stored in PostgreSQL.
 */
export class WebhookDeduplicationService {
  private static readonly DEDUP_TTL_HOURS = 24;

  /**
   * Check if this event has been processed recently
   * Returns true if duplicate (should skip processing)
   */
  async isDuplicate(event: WebhookEvent): Promise<boolean> {
    const hash = this.computeHash(event);

    const existing = await db
      .selectFrom('webhook_deduplication')
      .select('id')
      .where('hash', '=', hash)
      .where('created_at', '>', this.getTtlCutoff())
      .executeTakeFirst();

    if (existing) {
      logger.debug('Duplicate webhook detected', {
        deliveryId: event.deliveryId,
        hash: hash.substring(0, 8),
      });
      return true;
    }

    return false;
  }

  /**
   * Mark event as processed by storing its hash
   * Uses ON CONFLICT DO NOTHING for race condition safety
   */
  async markProcessed(event: WebhookEvent): Promise<void> {
    const hash = this.computeHash(event);

    await db
      .insertInto('webhook_deduplication')
      .values({
        id: crypto.randomUUID(),
        hash,
        delivery_id: event.deliveryId,
        event_type: event.eventType,
        created_at: new Date(),
      })
      .onConflict((oc) => oc.column('hash').doNothing())
      .execute();

    logger.debug('Marked webhook as processed', {
      deliveryId: event.deliveryId,
      hash: hash.substring(0, 8),
    });
  }

  /**
   * Clean up old deduplication entries beyond TTL
   * Call periodically (e.g., daily cron) to prevent table bloat
   */
  async cleanup(): Promise<number> {
    const result = await db
      .deleteFrom('webhook_deduplication')
      .where('created_at', '<', this.getTtlCutoff())
      .executeTakeFirst();

    const deleted = Number(result.numDeletedRows);
    if (deleted > 0) {
      logger.info('Cleaned up old deduplication entries', { deleted });
    }

    return deleted;
  }

  /**
   * Compute SHA-256 hash of immutable event fields
   * Hash includes: deliveryId, eventType, action, repo, issue/comment IDs
   */
  private computeHash(event: WebhookEvent): string {
    const data = JSON.stringify({
      deliveryId: event.deliveryId,
      eventType: event.eventType,
      action: event.action,
      repository: event.repository.fullName,
      issueNumber: event.issue?.number,
      commentId: event.comment?.id,
    });

    return crypto.createHash('sha256').update(data).digest('hex');
  }

  /**
   * Get cutoff timestamp for TTL window
   */
  private getTtlCutoff(): Date {
    const cutoff = new Date();
    cutoff.setHours(cutoff.getHours() - WebhookDeduplicationService.DEDUP_TTL_HOURS);
    return cutoff;
  }
}
```

```typescript
import crypto from 'crypto';
import db from '../persistence/config/database';
import { WebhookConfig, ProviderType } from './types';
import { createChildLogger } from '../config/logger';

const logger = createChildLogger({ component: 'WebhookConfigDAO' });

/**
 * Data access for webhook_configs table
 * Handles per-project webhook configuration
 */
export class WebhookConfigDAO {
  /**
   * Get webhook config for a project and provider type
   * Falls back to tenant-wide default (project_id IS NULL) if no project config
   */
  async getConfig(projectId: string, providerType: ProviderType): Promise<WebhookConfig | null> {
    // First try project-specific config
    let row = await db
      .selectFrom('webhook_configs')
      .selectAll()
      .where('project_id', '=', projectId)
      .where('provider_type', '=', providerType)
      .executeTakeFirst();

    // Fall back to tenant-wide default
    if (!row) {
      row = await db
        .selectFrom('webhook_configs')
        .selectAll()
        .where('project_id', 'is', null)
        .where('provider_type', '=', providerType)
        .executeTakeFirst();
    }

    if (!row) return null;

    return this.mapRowToConfig(row);
  }

  /**
   * Get webhook config by project ID only (for listing)
   */
  async getConfigsByProject(projectId: string): Promise<WebhookConfig[]> {
    const rows = await db
      .selectFrom('webhook_configs')
      .selectAll()
      .where('project_id', '=', projectId)
      .execute();

    return rows.map(row => this.mapRowToConfig(row));
  }

  /**
   * Create or update webhook config
   * Uses UPSERT pattern on (project_id, provider_type) unique constraint
   */
  async upsertConfig(config: Omit<WebhookConfig, 'id'>): Promise<WebhookConfig> {
    const now = new Date();
    const id = crypto.randomUUID();

    const result = await db
      .insertInto('webhook_configs')
      .values({
        id,
        project_id: config.projectId,
        provider_type: config.providerType,
        secret_key: config.secretKey,
        bot_username: config.botUsername,
        auto_execute: config.autoExecute,
        default_clanker_id: config.defaultClankerId,
        created_at: now,
        updated_at: now,
      })
      .onConflict((oc) =>
        oc.columns(['project_id', 'provider_type']).doUpdateSet({
          secret_key: config.secretKey,
          bot_username: config.botUsername,
          auto_execute: config.autoExecute,
          default_clanker_id: config.defaultClankerId,
          updated_at: now,
        })
      )
      .returningAll()
      .executeTakeFirstOrThrow();

    logger.info('Upserted webhook config', {
      projectId: config.projectId,
      providerType: config.providerType,
    });

    return this.mapRowToConfig(result);
  }

  /**
   * Delete webhook config
   */
  async deleteConfig(projectId: string, providerType: ProviderType): Promise<boolean> {
    const result = await db
      .deleteFrom('webhook_configs')
      .where('project_id', '=', projectId)
      .where('provider_type', '=', providerType)
      .executeTakeFirst();

    const deleted = Number(result.numDeletedRows) > 0;
    if (deleted) {
      logger.info('Deleted webhook config', { projectId, providerType });
    }

    return deleted;
  }

  /**
   * Find config by repository (for webhook routing)
   * Looks up project by repository_url, then gets webhook config
   */
  async getConfigByRepository(
    repoFullName: string,
    providerType: ProviderType
  ): Promise<{ config: WebhookConfig; projectId: string } | null> {
    // GitHub repos are formatted as owner/repo, repository_url is full URL
    const possibleUrls = [
      `https://github.com/${repoFullName}`,
      `https://github.com/${repoFullName}.git`,
      `git@github.com:${repoFullName}.git`,
    ];

    const project = await db
      .selectFrom('projects')
      .select(['id', 'repository_url'])
      .where('repository_url', 'in', possibleUrls)
      .executeTakeFirst();

    if (!project) {
      logger.debug('No project found for repository', { repoFullName });
      return null;
    }

    const config = await this.getConfig(project.id, providerType);
    if (!config) {
      logger.debug('No webhook config for project', { projectId: project.id, providerType });
      return null;
    }

    return { config, projectId: project.id };
  }

  private mapRowToConfig(row: any): WebhookConfig {
    return {
      id: row.id,
      projectId: row.project_id,
      providerType: row.provider_type,
      secretKey: row.secret_key,
      botUsername: row.bot_username,
      autoExecute: row.auto_execute,
      defaultClankerId: row.default_clanker_id,
    };
  }
}
```

```typescript
import crypto from 'crypto';
import db from '../persistence/config/database';
import { ProviderType } from './types';
import { createChildLogger } from '../config/logger';

const logger = createChildLogger({ component: 'WebhookFailedEventDAO' });

export interface FailedWebhookEvent {
  id: string;
  projectId: string | null;
  providerType: ProviderType;
  deliveryId: string | null;
  eventType: string | null;
  headers: Record<string, string>;
  payload: unknown;
  errorMessage: string | null;
  retryCount: number;
  createdAt: string;
  lastRetryAt: string | null;
}

/**
 * Data access for webhook_failed_events table
 * Stores failed webhook events for manual inspection and retry
 */
export class WebhookFailedEventDAO {
  /**
   * Store a failed webhook event
   */
  async storeFailedEvent(event: {
    projectId: string | null;
    providerType: ProviderType;
    deliveryId: string | null;
    eventType: string | null;
    headers: Record<string, string>;
    payload: unknown;
    errorMessage: string;
  }): Promise<string> {
    const id = crypto.randomUUID();

    await db
      .insertInto('webhook_failed_events')
      .values({
        id,
        project_id: event.projectId,
        provider_type: event.providerType,
        delivery_id: event.deliveryId,
        event_type: event.eventType,
        headers: JSON.stringify(event.headers),
        payload: JSON.stringify(event.payload),
        error_message: event.errorMessage,
        retry_count: 0,
        created_at: new Date(),
      })
      .execute();

    logger.info('Stored failed webhook event', {
      id,
      deliveryId: event.deliveryId,
      errorMessage: event.errorMessage,
    });

    return id;
  }

  /**
   * Get failed events for a project
   */
  async getFailedEvents(
    projectId: string | null,
    limit: number = 50
  ): Promise<FailedWebhookEvent[]> {
    let query = db
      .selectFrom('webhook_failed_events')
      .selectAll()
      .orderBy('created_at', 'desc')
      .limit(limit);

    if (projectId) {
      query = query.where('project_id', '=', projectId);
    }

    const rows = await query.execute();
    return rows.map(row => this.mapRowToEvent(row));
  }

  /**
   * Get a single failed event by ID
   */
  async getFailedEvent(id: string): Promise<FailedWebhookEvent | null> {
    const row = await db
      .selectFrom('webhook_failed_events')
      .selectAll()
      .where('id', '=', id)
      .executeTakeFirst();

    if (!row) return null;
    return this.mapRowToEvent(row);
  }

  /**
   * Increment retry count and update last retry timestamp
   */
  async markRetried(id: string): Promise<void> {
    await db
      .updateTable('webhook_failed_events')
      .set({
        retry_count: db.raw('retry_count + 1'),
        last_retry_at: new Date(),
      } as any)
      .where('id', '=', id)
      .execute();

    logger.debug('Marked failed event as retried', { id });
  }

  /**
   * Delete a failed event (after successful retry or manual dismissal)
   */
  async deleteFailedEvent(id: string): Promise<boolean> {
    const result = await db
      .deleteFrom('webhook_failed_events')
      .where('id', '=', id)
      .executeTakeFirst();

    const deleted = Number(result.numDeletedRows) > 0;
    if (deleted) {
      logger.info('Deleted failed webhook event', { id });
    }

    return deleted;
  }

  /**
   * Delete old failed events (cleanup)
   */
  async cleanupOldEvents(olderThanDays: number = 30): Promise<number> {
    const cutoff = new Date();
    cutoff.setDate(cutoff.getDate() - olderThanDays);

    const result = await db
      .deleteFrom('webhook_failed_events')
      .where('created_at', '<', cutoff)
      .executeTakeFirst();

    const deleted = Number(result.numDeletedRows);
    if (deleted > 0) {
      logger.info('Cleaned up old failed webhook events', { deleted, olderThanDays });
    }

    return deleted;
  }

  private mapRowToEvent(row: any): FailedWebhookEvent {
    return {
      id: row.id,
      projectId: row.project_id,
      providerType: row.provider_type,
      deliveryId: row.delivery_id,
      eventType: row.event_type,
      headers: typeof row.headers === 'string' ? JSON.parse(row.headers) : row.headers,
      payload: typeof row.payload === 'string' ? JSON.parse(row.payload) : row.payload,
      errorMessage: row.error_message,
      retryCount: row.retry_count,
      createdAt: row.created_at instanceof Date ? row.created_at.toISOString() : row.created_at,
      lastRetryAt: row.last_retry_at
        ? (row.last_retry_at instanceof Date ? row.last_retry_at.toISOString() : row.last_retry_at)
        : null,
    };
  }
}
```
