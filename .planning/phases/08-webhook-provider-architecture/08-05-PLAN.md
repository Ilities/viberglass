import { getWebhookProviderFactory } from './WebhookProviderFactory';
import { WebhookConfigDAO } from './WebhookConfigDAO';
import { ProviderConfig, ProviderType } from './types';
import { CredentialProviderFactory } from '../credentials/CredentialProviderFactory';
import db from '../persistence/config/database';
import { createChildLogger } from '../config/logger';

const logger = createChildLogger({ component: 'WebhookFeedbackService' });

export interface JobResultFeedback {
  jobId: string;
  ticketId: string;
  success: boolean;
  pullRequestUrl?: string;
  errorMessage?: string;
  executionTime?: number;
}

/**
 * Posts job results back to originating platform (GitHub issue, etc.)
 * Adds comments and updates labels based on job outcome
 */
export class WebhookFeedbackService {
  private providerFactory = getWebhookProviderFactory();
  private configDAO = new WebhookConfigDAO();
  private credentialProvider: CredentialProviderFactory;

  constructor(credentialProvider?: CredentialProviderFactory) {
    this.credentialProvider = credentialProvider || new CredentialProviderFactory();
  }

  /**
   * Post job result feedback to originating platform
   * Called after job completes (success or failure)
   */
  async postJobResult(feedback: JobResultFeedback): Promise<boolean> {
    try {
      // Get ticket metadata to find originating issue
      const ticket = await db
        .selectFrom('tickets')
        .select(['id', 'project_id', 'metadata', 'ticket_system', 'external_ticket_id'])
        .where('id', '=', feedback.ticketId)
        .executeTakeFirst();

      if (!ticket) {
        logger.warn('Ticket not found for feedback', { ticketId: feedback.ticketId });
        return false;
      }

      const metadata = typeof ticket.metadata === 'string'
        ? JSON.parse(ticket.metadata)
        : ticket.metadata;

      // Check if this ticket came from a webhook
      if (metadata?.source !== 'webhook') {
        logger.debug('Ticket not from webhook, skipping feedback', { ticketId: feedback.ticketId });
        return false;
      }

      const providerType = (metadata.provider || ticket.ticket_system) as ProviderType;
      const issueNumber = metadata.issueNumber || parseInt(ticket.external_ticket_id || '0');
      const repository = metadata.repository as string;

      if (!repository || !issueNumber) {
        logger.warn('Missing repository or issue number for feedback', {
          ticketId: feedback.ticketId,
          repository,
          issueNumber,
        });
        return false;
      }

      // Get webhook config for credentials
      const configResult = await this.configDAO.getConfigByRepository(repository, providerType);
      if (!configResult) {
        logger.warn('No webhook config for repository', { repository });
        return false;
      }

      const { config } = configResult;

      // Get GitHub token from credentials
      // Use a separate credential key for API access (not webhook secret)
      const [owner, repo] = repository.split('/');
      const token = await this.credentialProvider.get('default', `${owner}_${repo}_GITHUB_TOKEN`)
        || await this.credentialProvider.get('default', 'GITHUB_TOKEN');

      if (!token) {
        logger.warn('No GitHub token found for feedback', { repository });
        return false;
      }

      const providerConfig: ProviderConfig = { owner, repo, token };
      const provider = this.providerFactory.getProvider(providerType);

      // Post comment with result
      const comment = this.formatResultComment(feedback);
      await provider.postComment(providerConfig, issueNumber, comment);

      // Update labels
      const { add, remove } = this.getLabelsForResult(feedback);
      if (add.length > 0 || remove.length > 0) {
        await provider.updateLabels(providerConfig, issueNumber, add, remove);
      }

      logger.info('Posted job result feedback', {
        jobId: feedback.jobId,
        ticketId: feedback.ticketId,
        repository,
        issueNumber,
        success: feedback.success,
      });

      return true;
    } catch (error) {
      logger.error('Failed to post job result feedback', {
        error: (error as Error).message,
        jobId: feedback.jobId,
        ticketId: feedback.ticketId,
      });
      return false;
    }
  }

  /**
   * Format comment body for job result
   */
  private formatResultComment(feedback: JobResultFeedback): string {
    if (feedback.success) {
      let comment = `## Viberator Fix Submitted\n\n`;
      comment += `A fix has been automatically generated for this issue.\n\n`;

      if (feedback.pullRequestUrl) {
        comment += `**Pull Request:** ${feedback.pullRequestUrl}\n\n`;
      }

      if (feedback.executionTime) {
        const seconds = Math.round(feedback.executionTime / 1000);
        comment += `*Completed in ${seconds} seconds*\n`;
      }

      comment += `\n---\n*Automated by [Viberator](https://github.com/your-org/viberator)*`;
      return comment;
    } else {
      let comment = `## Viberator Fix Failed\n\n`;
      comment += `Unable to automatically fix this issue.\n\n`;

      if (feedback.errorMessage) {
        comment += `**Error:** ${feedback.errorMessage}\n\n`;
      }

      comment += `Please review the issue manually or provide more context.\n`;
      comment += `\n---\n*Automated by [Viberator](https://github.com/your-org/viberator)*`;
      return comment;
    }
  }

  /**
   * Determine labels to add/remove based on result
   */
  private getLabelsForResult(feedback: JobResultFeedback): { add: string[]; remove: string[] } {
    if (feedback.success) {
      return {
        add: ['viberator:fix-submitted'],
        remove: ['viberator:pending', 'viberator:failed'],
      };
    } else {
      return {
        add: ['viberator:failed'],
        remove: ['viberator:pending', 'viberator:fix-submitted'],
      };
    }
  }
}

const API_BASE = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:8888';

export interface WebhookConfig {
  id: string;
  projectId: string | null;
  providerType: 'github' | 'gitlab' | 'bitbucket';
  secretKey: string;
  botUsername: string | null;
  autoExecute: boolean;
  defaultClankerId: string | null;
}

export interface WebhookSetupInstructions {
  title: string;
  steps: string[];
  payloadUrl: string;
  contentType: string;
  events: string[];
  secretInstructions: string;
}

export interface SaveWebhookConfigRequest {
  secretKey: string;
  botUsername?: string;
  autoExecute?: boolean;
  defaultClankerId?: string;
}

/**
 * Get webhook configurations for a project
 */
export async function getWebhookConfigs(projectId: string): Promise<WebhookConfig[]> {
  const response = await fetch(`${API_BASE}/api/webhooks/config/${projectId}`);
  if (!response.ok) {
    throw new Error('Failed to fetch webhook configs');
  }
  const data = await response.json();
  return data.configs;
}

/**
 * Save webhook configuration
 */
export async function saveWebhookConfig(
  projectId: string,
  provider: string,
  config: SaveWebhookConfigRequest
): Promise<WebhookConfig> {
  const response = await fetch(`${API_BASE}/api/webhooks/config/${projectId}/${provider}`, {
    method: 'PUT',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(config),
  });

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.error || 'Failed to save webhook config');
  }

  const data = await response.json();
  return data.config;
}

/**
 * Delete webhook configuration
 */
export async function deleteWebhookConfig(
  projectId: string,
  provider: string
): Promise<void> {
  const response = await fetch(`${API_BASE}/api/webhooks/config/${projectId}/${provider}`, {
    method: 'DELETE',
  });

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.error || 'Failed to delete webhook config');
  }
}

/**
 * Get setup instructions for a provider
 */
export async function getSetupInstructions(provider: string): Promise<WebhookSetupInstructions> {
  const response = await fetch(`${API_BASE}/api/webhooks/setup/${provider}`);
  if (!response.ok) {
    throw new Error('Failed to fetch setup instructions');
  }
  const data = await response.json();
  return data.instructions;
}

/**
 * Generate a random webhook secret
 */
export function generateWebhookSecret(): string {
  const array = new Uint8Array(32);
  crypto.getRandomValues(array);
  return Array.from(array, byte => byte.toString(16).padStart(2, '0')).join('');
}

'use client';

import { useState, useEffect } from 'react';
import {
  getWebhookConfigs,
  saveWebhookConfig,
  deleteWebhookConfig,
  generateWebhookSecret,
  WebhookConfig,
} from '@/lib/api/webhooks';
import { getClankers } from '@/lib/api/clankers';
import { toast } from 'sonner';

interface Props {
  projectId: string;
  provider: 'github' | 'gitlab' | 'bitbucket';
}

export function WebhookConfigForm({ projectId, provider }: Props) {
  const [config, setConfig] = useState<WebhookConfig | null>(null);
  const [clankers, setClankers] = useState<Array<{ id: string; name: string }>>([]);
  const [loading, setLoading] = useState(true);
  const [saving, setSaving] = useState(false);

  // Form state
  const [secretKey, setSecretKey] = useState('');
  const [botUsername, setBotUsername] = useState('');
  const [autoExecute, setAutoExecute] = useState(false);
  const [defaultClankerId, setDefaultClankerId] = useState('');

  useEffect(() => {
    loadData();
  }, [projectId, provider]);

  async function loadData() {
    try {
      setLoading(true);
      const [configs, clankersData] = await Promise.all([
        getWebhookConfigs(projectId),
        getClankers(),
      ]);

      const existingConfig = configs.find(c => c.providerType === provider);
      if (existingConfig) {
        setConfig(existingConfig);
        setSecretKey(existingConfig.secretKey);
        setBotUsername(existingConfig.botUsername || '');
        setAutoExecute(existingConfig.autoExecute);
        setDefaultClankerId(existingConfig.defaultClankerId || '');
      }

      setClankers(clankersData);
    } catch (error) {
      toast.error('Failed to load webhook configuration');
    } finally {
      setLoading(false);
    }
  }

  async function handleSave() {
    if (!secretKey) {
      toast.error('Secret key is required');
      return;
    }

    try {
      setSaving(true);
      await saveWebhookConfig(projectId, provider, {
        secretKey,
        botUsername: botUsername || undefined,
        autoExecute,
        defaultClankerId: defaultClankerId || undefined,
      });
      toast.success('Webhook configuration saved');
      loadData();
    } catch (error) {
      toast.error('Failed to save webhook configuration');
    } finally {
      setSaving(false);
    }
  }

  async function handleDelete() {
    if (!confirm('Are you sure you want to delete this webhook configuration?')) {
      return;
    }

    try {
      setSaving(true);
      await deleteWebhookConfig(projectId, provider);
      toast.success('Webhook configuration deleted');
      setConfig(null);
      setSecretKey('');
      setBotUsername('');
      setAutoExecute(false);
      setDefaultClankerId('');
    } catch (error) {
      toast.error('Failed to delete webhook configuration');
    } finally {
      setSaving(false);
    }
  }

  function handleGenerateSecret() {
    const secret = generateWebhookSecret();
    setSecretKey(secret);
    toast.success('Secret generated - save to apply');
  }

  if (loading) {
    return <div className="animate-pulse h-48 bg-gray-100 rounded" />;
  }

  return (
    <div className="space-y-4">
      <div>
        <label className="block text-sm font-medium text-gray-700 mb-1">
          Webhook Secret
        </label>
        <div className="flex gap-2">
          <input
            type="text"
            value={secretKey}
            onChange={(e) => setSecretKey(e.target.value)}
            placeholder="Enter or generate a secret"
            className="flex-1 rounded-md border-gray-300 shadow-sm focus:border-blue-500 focus:ring-blue-500"
          />
          <button
            type="button"
            onClick={handleGenerateSecret}
            className="px-3 py-2 bg-gray-100 text-gray-700 rounded-md hover:bg-gray-200"
          >
            Generate
          </button>
        </div>
        <p className="mt-1 text-xs text-gray-500">
          This secret is used to verify webhook signatures
        </p>
      </div>

      <div>
        <label className="block text-sm font-medium text-gray-700 mb-1">
          Bot Username (for @mentions)
        </label>
        <input
          type="text"
          value={botUsername}
          onChange={(e) => setBotUsername(e.target.value)}
          placeholder="viberator"
          className="w-full rounded-md border-gray-300 shadow-sm focus:border-blue-500 focus:ring-blue-500"
        />
        <p className="mt-1 text-xs text-gray-500">
          Only process comments that @mention this username
        </p>
      </div>

      <div className="flex items-center gap-2">
        <input
          type="checkbox"
          id="autoExecute"
          checked={autoExecute}
          onChange={(e) => setAutoExecute(e.target.checked)}
          className="rounded border-gray-300 text-blue-600 focus:ring-blue-500"
        />
        <label htmlFor="autoExecute" className="text-sm text-gray-700">
          Auto-execute clanker when webhook creates ticket
        </label>
      </div>

      {autoExecute && (
        <div>
          <label className="block text-sm font-medium text-gray-700 mb-1">
            Default Clanker
          </label>
          <select
            value={defaultClankerId}
            onChange={(e) => setDefaultClankerId(e.target.value)}
            className="w-full rounded-md border-gray-300 shadow-sm focus:border-blue-500 focus:ring-blue-500"
          >
            <option value="">Select a clanker...</option>
            {clankers.map((clanker) => (
              <option key={clanker.id} value={clanker.id}>
                {clanker.name}
              </option>
            ))}
          </select>
        </div>
      )}

      <div className="flex gap-2 pt-4">
        <button
          onClick={handleSave}
          disabled={saving}
          className="px-4 py-2 bg-blue-600 text-white rounded-md hover:bg-blue-700 disabled:opacity-50"
        >
          {saving ? 'Saving...' : 'Save Configuration'}
        </button>
        {config && (
          <button
            onClick={handleDelete}
            disabled={saving}
            className="px-4 py-2 bg-red-100 text-red-700 rounded-md hover:bg-red-200 disabled:opacity-50"
          >
            Delete
          </button>
        )}
      </div>
    </div>
  );
}

'use client';

import { useState, useEffect } from 'react';
import { getSetupInstructions, WebhookSetupInstructions as Instructions } from '@/lib/api/webhooks';
import { toast } from 'sonner';

interface Props {
  provider: 'github' | 'gitlab' | 'bitbucket';
}

export function WebhookSetupInstructions({ provider }: Props) {
  const [instructions, setInstructions] = useState<Instructions | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadInstructions();
  }, [provider]);

  async function loadInstructions() {
    try {
      setLoading(true);
      const data = await getSetupInstructions(provider);
      setInstructions(data);
    } catch (error) {
      toast.error('Failed to load setup instructions');
    } finally {
      setLoading(false);
    }
  }

  function copyToClipboard(text: string, label: string) {
    navigator.clipboard.writeText(text);
    toast.success(`${label} copied to clipboard`);
  }

  if (loading) {
    return <div className="animate-pulse h-48 bg-gray-100 rounded" />;
  }

  if (!instructions) {
    return <div className="text-gray-500">Failed to load instructions</div>;
  }

  return (
    <div className="space-y-4">
      <h3 className="font-medium text-gray-900">{instructions.title}</h3>

      <div className="space-y-2">
        {instructions.steps.map((step, index) => (
          <p key={index} className="text-sm text-gray-600">
            {step}
          </p>
        ))}
      </div>

      <div className="mt-4 space-y-3">
        <div>
          <label className="block text-xs font-medium text-gray-500 uppercase mb-1">
            Payload URL
          </label>
          <div className="flex gap-2">
            <code className="flex-1 block p-2 bg-gray-100 rounded text-sm font-mono break-all">
              {instructions.payloadUrl}
            </code>
            <button
              onClick={() => copyToClipboard(instructions.payloadUrl, 'URL')}
              className="px-2 py-1 text-xs bg-gray-200 rounded hover:bg-gray-300"
            >
              Copy
            </button>
          </div>
        </div>

        <div>
          <label className="block text-xs font-medium text-gray-500 uppercase mb-1">
            Content Type
          </label>
          <code className="block p-2 bg-gray-100 rounded text-sm font-mono">
            {instructions.contentType}
          </code>
        </div>

        <div>
          <label className="block text-xs font-medium text-gray-500 uppercase mb-1">
            Events to Subscribe
          </label>
          <div className="flex flex-wrap gap-2">
            {instructions.events.map((event) => (
              <span
                key={event}
                className="px-2 py-1 bg-blue-100 text-blue-800 text-xs rounded"
              >
                {event}
              </span>
            ))}
          </div>
        </div>

        <div className="mt-4 p-3 bg-yellow-50 border border-yellow-200 rounded-md">
          <p className="text-sm text-yellow-800">
            <strong>Important:</strong> {instructions.secretInstructions}
          </p>
        </div>
      </div>
    </div>
  );
}
