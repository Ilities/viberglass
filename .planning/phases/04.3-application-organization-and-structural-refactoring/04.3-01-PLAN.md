       - On error: return 400 with { error: "Validation error", details: [...] }
       - On success: replace req.body with validated value, call next()
    3. Keep the error format identical to current implementation for backward compatibility
    4. The details array should contain { field, message } objects

    Pattern to follow:
    ```typescript
    import { Request, Response, NextFunction } from "express";
    import Joi from "joi";

    function createValidator(schema: Joi.Schema) {
      return (req: Request, res: Response, next: NextFunction) => {
        const { error, value } = schema.validate(req.body);
        if (error) {
          return res.status(400).json({
            error: "Validation error",
            details: error.details.map((detail) => ({
              field: detail.path.join("."),
              message: detail.message,
            })),
          });
        }
        req.body = value;
        next();
      };
    }
    ```

    5. Keep validateUuidParam, validateFileUploads as-is (they use custom logic)
    6. Replace all schema-specific validators with calls to createValidator()
    7. Keep the existing export names for backward compatibility with route files
