  <files>.github/workflows/frontend-ci.yml</files>
  <action>
    Create .github/workflows/frontend-ci.yml that:
    1. Triggers on pull_request with paths: platform/frontend/**, packages/types/**
    2. Runs on ubuntu-latest
    3. Checks out code (actions/checkout@v4)
    4. Sets up Node.js (actions/setup-node@v4) with node-version-file: package.json
    5. Caches npm modules with cache: 'npm'
    6. Installs dependencies: npm ci --legacy-peer-deps
    7. Runs lint: npm run lint -w @viberator/frontend
    8. Runs tests: npm run test:unit -w @viberator/frontend (if tests exist)
    9. Runs build to verify SSR: npm run build -w @viberator/frontend

    Build step should set NEXT_PUBLIC_API_URL to a placeholder.
  </action>

<task type="auto">
  <name>Task 2: Create dev deployment workflow (auto-deploy on main)</name>
  <files>.github/workflows/deploy-frontend-dev.yml</files>
  <action>
    Create .github/workflows/deploy-frontend-dev.yml that:
    1. Triggers on push to main branch with paths: platform/frontend/**, packages/types/**
    2. Runs on ubuntu-latest with permissions: id-token: write, contents: read
    3. Checks out code
    4. Sets up Node.js with npm cache
    5. Installs dependencies: npm ci --legacy-peer-deps
    6. Configures AWS credentials via OIDC:
       - uses: aws-actions/configure-aws-credentials@v4
       - role-to-assume: ${{ secrets.AWS_ROLE_ARN }}
       - aws-region: eu-west-1
    7. Fetches backend API URL from SSM parameter (/viberator/dev/frontend/apiUrl)
    8. Builds frontend with environment variables:
       - NEXT_PUBLIC_API_URL: from SSM parameter
       - npm run build -w @viberator/frontend
    9. Triggers Amplify deployment:
       - aws amplify start-deployment --app-id ${{ secrets.DEV_AMPLIFY_APP_ID }} --branch-name ${{ secrets.DEV_AMPLIFY_BRANCH }}
    10. Waits for deployment completion with polling loop:
        - aws amplify get-job --job-id <job_id> until status is SUCCEED
        - Timeout: 10 minutes
    11. Outputs deployment URL for GitHub deployment UI

    Environment-specific: dev Amplify app ID and branch
  </action>
  <verify>
    cat .github/workflows/deploy-frontend-dev.yml
  </verify>
  <done>
    Dev frontend deployment triggers on main push and deploys to Amplify
  </done>
</task>
