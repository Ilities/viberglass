    4. build: npm run build
    5. Configure artifacts: baseDirectory: .next, files: **/*

    Template:
    ```yaml

    frontend:
      phases:
        preBuild:
          commands:
            - npm ci --legacy-peer-deps
        build:
          commands:
            - npm run build
      artifacts:
        baseDirectory: .next
        files:
          - '**/*'
      cache:
        paths:
          - .npm/**/*
          - next/cache/**/*
    ```
